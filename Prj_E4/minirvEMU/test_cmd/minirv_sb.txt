# sb 指令测试程序
# 验证字节存储功能

_start:
    # 设置基地址 (假设数据区从地址 0x1000 开始)
    lui x1, 0x00001        # x1 = 0x00001000
    
    # 测试1: 使用 lw 读取初始值
    lw x2, 0(x1)           # 从内存地址 0x1000 加载整个字到 x2
    
    # 测试2: 准备要存储的字节值
    addi x3, x0, 0xEF      # x3 = 0xEF (将要存储到地址 a+0)
    addi x4, x0, 0xCD      # x4 = 0xCD (将要存储到地址 a+1)
    addi x5, x0, 0xAB      # x5 = 0xAB (将要存储到地址 a+2)
    addi x6, x0, 0x90      # x6 = 0x90 (将要存储到地址 a+3)
    
    # 测试3: 使用 sb 指令分别写入4个字节
    sb x3, 0(x1)           # 将 0xEF 存储到内存地址 0x1000
    sb x4, 1(x1)           # 将 0xCD 存储到内存地址 0x1001
    sb x5, 2(x1)           # 将 0xAB 存储到内存地址 0x1002
    sb x6, 3(x1)           # 将 0x90 存储到内存地址 0x1003
    
    # 测试4: 使用 lw 读取修改后的值
    lw x7, 0(x1)           # 从内存地址 0x1000 加载整个字到 x7
    
    # 使用 jalr 实现无限循环
    addi x8, x0, 0x2C      # x8 = 循环地址 0x2C
    jalr x0, 0(x8)         # 跳转到地址 0x2C，形成无限循环


指令编码
