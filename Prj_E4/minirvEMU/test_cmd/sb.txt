0x00: lui   x1, 0x00010      # x1 = 0x00010000 (基地址 a)

# 写入初始值 0x12345678
0x04: lui   x2, 0x12345
0x08: addi  x2, x2, 0x678
0x0C: sw    x2, 0(x1)

# 验证 lw
0x10: lw    x3, 0(x1)        # x3 = 0x12345678

# 构造要写入的 byte
0x14: addi  x4, x0, 0xef     # x4 = 0xef
0x18: addi  x5, x0, 0xcd     # x5 = 0xcd
0x1C: addi  x6, x0, 0xab     # x6 = 0xab
0x20: addi  x7, x0, 0x90     # x7 = 0x90

# 按 byte 写入（注意小端）
0x24: sb    x4, 0(x1)        # a+0 = ef
0x28: sb    x5, 1(x1)        # a+1 = cd
0x2C: sb    x6, 2(x1)        # a+2 = ab
0x30: sb    x7, 3(x1)        # a+3 = 90

# 再次读取
0x34: lw    x8, 0(x1)        # 预期 x8 = 0x90abcdef

# 死循环
0x38: jalr  x0, x0, 0x38
